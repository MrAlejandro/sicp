```(define (f n)
    (if (< n 3)
        n
        (+ (f (- n 1)) (f (- n 2)) (f (- n 3)))
    )
)```

```(f 6)```

```(define (f-iterative n)
    (define (iter a b c count) 
        (cond ((< count 0) count)
              ((= count 0) c)
              (else (iter (+ a b c) a b (- count 1)))
        )
    )
    (iter 2 1 0 n)
)```

```(f-iterative 6)```
