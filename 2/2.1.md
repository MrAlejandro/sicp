(define (make-rat n d) (cons n d))
(define (numer n) (car n))
(define (denom d) (cdr d))

(define (add-rat x y)
    (make-rat
        (+
            (* (numer x) (denom y))
            (* (numer y) (denom x))
        )
        (* (denom x) (denom y))
    )
)

(define (sub-rat x y)
    (make-rat
        (-
            (* (numer x) (denom y))
            (* (numer y) (denom x))
        )
        (* (denom x) (denom y))
    )
)

(define (mul-rat x y)
    (make-rat
        (* (numer x) (numer y))
        (* (denom x) (denom y))
    )
)

(define (div-rat x y)
    (make-rat
        (* (numer x) (denom y))
        (* (denom x) (numer y))
    )
)

(define (equal-rat? x y)
    (=
        (* (numer x) (denom y))
        (* (numer y) (denom x))
    )
)

(define (print-rat x)
    (newline)
    (display (numer x))
    (display "/")
    (display (denom x))
)

(define one-half (make-rat 1 2))
(define one-third (make-rat 1 3))

(print-rat (add-rat one-half one-third))

(define (mul-rat n d)
    (let (
            (g ((if (< d 0) - +) 1))
        )
        (make-rat (/ n g) (/ d g))
    )
)

(define test (mul-rat 1 -2))
(print-rat test)
